144  template <typename T>
145  void ArrayCopy<T>::toPy (void* to, const T* from, size_t nr)
146  {
147  if (sizeof(T) == sizeof(typename TypeConvTraits<T>::python_type)) {
148  ::memcpy (to, from, nr*sizeof(T));
149  } else {
150  typename TypeConvTraits<T>::python_type* dst =
151  static_cast<typename TypeConvTraits<T>::python_type*>(to);
152  for (size_t i=0; i<nr; i++) {
153  dst[i] = from[i];
154  }
155  }
156  }
157  template <typename T>
158  void ArrayCopy<T>::fromPy (T* to, const void* from, size_t nr)
159  {
160  if (sizeof(T) == sizeof(typename TypeConvTraits<T>::python_type)) {
161  ::memcpy (to, from, nr*sizeof(T));
162  } else {
163  const typename TypeConvTraits<T>::python_type* src =
164  static_cast<const typename TypeConvTraits<T>::python_type*>(from);
165  for (size_t i=0; i<nr; i++) {
166  to[i] = src[i];
167  }
168  }

170  template <typename T>
171  Array<T> ArrayCopy<T>::toArray (const IPosition& shape,
172  void* data, bool copy)
173  {
174  // If the python array was contiguous, etc., we can directly use
175  // its data because the Array used is only temporary.
176  // However, if a copy of the Python array was made in PycArray.cc,
177  // we cannot do that because the Python copy is out of scope when
178  // the Array object gets used.
179  if (!copy) {
180  if (sizeof(T) == sizeof(typename TypeConvTraits<T>::python_type)) {
181  return Array<T> (shape, static_cast<T*>(data), SHARE);
182  }
183  }
184  Array<T> arr(shape);
185  fromPy (arr.data(), data, arr.size());
186  return arr;
187  }

276  ValueHolder makeArray (PyObject* obj_ptr, Bool copyData)
277  {
278  if (! PycArrayCheck(obj_ptr)) {
279  throw AipsError ("PycArray: python object is not an array");
280  }
281  PyArrayObject* po = (PyArrayObject*)obj_ptr;
282  boost::python::object obj;
283  bool docopy = copyData; // copy data if wanted or needed
284  if (! PyArray_ISCONTIGUOUS(po)
285  || ! PyArray_ISALIGNED(po)
286  || PyArray_ISBYTESWAPPED(po)) {
287  boost::python::handle<> py_hdl(obj_ptr);
288  boost::python::object py_obj(py_hdl);
289  // incr refcount, because ~object decrements it
290  boost::python::incref(obj_ptr);
291  obj = py_obj.attr("copy")();
292  po = (PyArrayObject*)(obj.ptr());
293  docopy = true;
294  }
295  // Swap axes, because Casacore has row minor and Python row major order.
296  // A scalar is treated as a vector with length 1.
297  int nd = PyArray_NDIM(po);
298  IPosition shp(1, 1);
299  if (nd > 0) {
300  shp.resize (nd);
301  for (int i=0; i<nd; i++) {
302  shp[i] = PyArray_DIMS(po)[nd-i-1];
303  }
304  }
305  // Assert array is contiguous now.
306  // If the array is empty, numarray still sees it as non-contiguous.
307  if (shp.product() > 0) {
308  AlwaysAssert (PyArray_ISCONTIGUOUS(po)
310  && !PyArray_ISBYTESWAPPED(po), AipsError);
311  }
312  // Create the correct array.
313  switch (PyArray_TYPE(po)) {
314  case NPY_BOOL:
315  return ValueHolder (ArrayCopy<Bool>::toArray(shp, PyArray_DATA(po), docopy));
316  case NPY_INT16:
317  return ValueHolder (ArrayCopy<Short>::toArray(shp, PyArray_DATA(po), docopy));
318  case NPY_UINT16:
319  return ValueHolder (ArrayCopy<uShort>::toArray(shp, PyArray_DATA(po), docopy));
320  case NPY_INT32:
321  return ValueHolder (ArrayCopy<Int>::toArray(shp, PyArray_DATA(po), docopy));
322  case NPY_UINT32:
323  return ValueHolder (ArrayCopy<uInt>::toArray(shp, PyArray_DATA(po), docopy));
324  case NPY_INT64:
325  return ValueHolder (ArrayCopy<Int64>::toArray(shp, PyArray_DATA(po), docopy));
326  case NPY_FLOAT32:
327  return ValueHolder (ArrayCopy<Float>::toArray(shp, PyArray_DATA(po), docopy));
328  case NPY_FLOAT64:
329  return ValueHolder (ArrayCopy<Double>::toArray(shp, PyArray_DATA(po), docopy));
330  case NPY_COMPLEX64:
331  return ValueHolder (ArrayCopy<Complex>::toArray(shp, PyArray_DATA(po), docopy));
332  case NPY_COMPLEX128:
333  return ValueHolder (ArrayCopy<DComplex>::toArray(shp, PyArray_DATA(po), docopy));
334  case NPY_OBJECT:
335  return ValueHolder (ArrayCopy<String>::toArray(shp, PyArray_DATA(po), docopy));
336  default:
337  // Some types can be the same as other types, so they cannot
338  // be used in the switch (compiler complains).
339  // This is true for BYTE and SBYTE which can equal to BOOL in numarray.
340  // Similarly for STRING which exists for numpy and is set to
341  // INT for numarray.
342  if (PyArray_TYPE(po) == NPY_UINT64) {
343  Array<uInt64> arr = ArrayCopy<uInt64>::toArray(shp, PyArray_DATA(po), False);
344  Array<Int64> res(arr.shape());
345  convertArray (res, arr);
346  return ValueHolder(res);
347  } else if (PyArray_TYPE(po) == NPY_INT8) {
348  Array<Char> arr = ArrayCopy<Char>::toArray(shp, PyArray_DATA(po), False);
349  Array<Short> res(arr.shape());
350  convertArray (res, arr);
351  return ValueHolder(res);
352  } else if (PyArray_TYPE(po) == NPY_UINT8) {
353  // Copy using Char, because uChar is mapped to Short in the Traits.
354  Array<Char> arr = ArrayCopy<Char>::toArray(shp, PyArray_DATA(po), False);
355  Array<Short> res(arr.shape());
356  void* varr = &arr;
357  Array<uChar>* uarr = static_cast<Array<uChar>*>(varr);
358  convertArray (res, *uarr);
359  return ValueHolder(res);
360  } else if (PyArray_TYPE(po) == NPY_STRING) {
361  size_t slen = 0;
362  if (nd > 0) {
363  slen = PyArray_STRIDES(po)[nd-1];
364  }
365  return ValueHolder (ArrayCopyStr_toArray(shp, PyArray_DATA(po), slen));
366  }
367  break;
368  }
369  throw AipsError ("PycArray: unknown python array data type");
370  }
